SELECT *
FROM (SELECT * 
FROM (SELECT grommet_product_categories.product_id
FROM grommet_gifts_categories
INNER JOIN grommet_product_categories ON grommet_gifts_categories.id=grommet_product_categories.product_category_id
WHERE grommet_gifts_categories.sub_category="Beauty & Personal Care" OR grommet_gifts_categories.sub_category="Skincare" OR grommet_gifts_categories.sub_category="Jewelry") AS catagory_id
INNER JOIN grommet_products ON catagory_id.product_id =grommet_products.id
WHERE grommet_products.is_sold_out="0"
) AS A,
(SELECT * 
FROM (SELECT grommet_product_to_keyword.product_id
FROM grommet_product_keywords
INNER JOIN grommet_product_to_keyword ON grommet_product_keywords.id=grommet_product_to_keyword.keyword_id
WHERE grommet_product_keywords.keyword="Aromatherapy" OR grommet_product_keywords.keyword="Hair accessor") AS keyword_id
INNER JOIN grommet_products ON keyword_id.product_id=grommet_products.id
WHERE grommet_products.is_sold_out="0"
) AS B
